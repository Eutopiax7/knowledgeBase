# 题

取{2,3,1,2,4,3}中>=7的最短子串

# 答案

```
#include <iostream>  
using namespace std;  
  
int main(){  
long long sum=0;  
int minn=1e9;  
int n;  
int a[105];  
int s;  
cin>>n>>s;  
for(int i=1;i<=n;i++){  
cin>>a[i];  
}  
for(int i=0,j=0;i<=n;i++){  
while(sum<s&&j<=n){  
sum+=a[j];  
j++;  
}  
if(sum>s){  
minn=min(minn,j-i);  
}  
sum-=a[i];  
}  
cout<<(minn==1e9?0:minn)<<endl;  
return 0;  
}
```
